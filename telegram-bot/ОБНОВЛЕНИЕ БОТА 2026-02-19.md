```python
async def pack_status(update: Update, context: ContextTypes.DEFAULT_TYPE):
    """Команда /pack - статус Pack"""
    content_dir = REPO_PATH / "content"

    # Подсчёт документов по семействам
    families = {}
    for family in range(10):  # F0-F9
        family_docs = list(content_dir.rglob(f"*F{family}*"))
        families[f"F{family}"] = len(family_docs)

    response = "📦 Статус Pack (VK-offee)\n\n"
    response += "Документов по семействам:\n"
    for family, count in families.items():
        response += f"{family}: {count} документов\n"

    await update.message.reply_text(response)
```
